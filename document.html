document.addEventListener('DOMContentLoaded', function () { const studentInfoData = JSON.parse(localStorage.getItem('studentInfo')); if (!studentInfoData) { window.location.href = 'index.html'; } const studentId = studentInfoData ? studentInfoData.id : null; document.querySelector('.realname').textContent = studentInfoData.realname; const logoutButton = document.getElementById('logoutButton'); if (logoutButton) { logoutButton.addEventListener('click', function (e) { e.preventDefault(); // 防止链接默认的跳转行为 localStorage.clear(); // 清除 localStorage window.location.href = 'index.html'; // 跳转到 index.html }); } });
